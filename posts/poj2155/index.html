#include <cstdio>
#include <cstring>

using namespace std;

const int MAX_N = 1000;
int bit[MAX_N + 10][MAX_N + 10];
int N, T;

void add(int a, int b){
    for(int i = a; i <= N; i += (i & -i)){
        for(int j = b; j <= N; j += (j & -j)){
            bit[i][j]++;
        }
    }
}

int sum(int a, int b){
    int s = 0;
    for(int i = a; i > 0; i -= (i & -i)){
        for(int j = b; j > 0; j -= (j & -j)){
            s += bit[i][j];
        }
    }
    return s;
}

int main(){
    int TC;
    scanf("%d\n", &TC);
    while(TC--){
        memset(bit, 0, sizeof(bit));
        scanf("%d %d\n", &N, &T);
        for(int i = 0; i < T; i++){
            char c[3];
            scanf("%s", c);
            if(c[0] == 'C'){
                int x1, y1, x2, y2;
                scanf("%d %d %d %d", &x1, &y1, &x2, &y2);
                add(x1, y1);
                add(x2 + 1, y1);
                add(x1, y2 + 1);
                add(x2 + 1, y2 + 1);
            }
            else{
                int x, y;
                scanf("%d %d", &x, &y);
                printf("%d\n", sum(x, y) % 2);
            }

        }
        puts("");
    }
    return 0;
}
