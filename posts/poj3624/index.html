POJ 3624 Charm Bracelet

Jan 18, 2017
1 min read
POJ DP Knapsack

題解

01背包問題裸題, 但需要空間優化

作法

dp陣列開成兩條一維, 交替使用

AC code

#include <cstdio>
#include <algorithm>

using namespace std;

int n, m;
int dp[2][13000];
int w[3500], d[3500];

int main(){
    scanf("%d %d", &n, &m);
    for(int i = 0; i < n; i++){
        scanf("%d %d", &w[i], &d[i]);
    }

    for(int i = 0; i < n; i++){
        for(int j = 0; j <= m; j++){
            if(j < w[i])
                dp[(i + 1) & 1][j] = dp[i & 1][j];
            else
                dp[(i + 1) & 1][j] = max(dp[i & 1][j], dp[i & 1][j - w[i]] + d[i]);
        }
    }
    printf("%d\n", dp[n & 1][m]);
    return 0;
}
