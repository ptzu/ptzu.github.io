typedef pair <int,int> pii;

#define pb push_back
#define sz(a) (int)a.size()
#define INF 0x3f3f3f3f
#define st first
#define nd second
#define DEBUG(x) printf("%d\n", x), exit(0);
// LLONG_MIN LLONG_MAX INT_MIN INT_MAX

int s, x1, x2, v1, v2, p, d;

int main(){
    scanf("%d %d %d %d %d %d %d", &s, &x1, &x2, &v1, &v2, &p, &d);
    //opposite direction
    int dis = 0;
    int ans = abs(x2 - x1) * v2;
    if((x2 - x1) * d < 0){
        if(d > 0)
            dis = s - p + abs(x2 - s);
        else
            dis = p + x2;
    }
    else{
        if(d > 0 && p > x1)
            dis = (s - p) + s + x2;
        else if(d < 0 && p < x1)
            dis = p + s + abs(s - x2);
        else
            dis = abs(x2 - p);
    }
    printf("%d\n", min(ans, dis * v1));
    return 0;
}

#include <bits/stdc++.h>

using namespace std;

typedef pair <int,int> pii;
typedef long long ll;

#define pb push_back
#define sz(a) (int)a.size()
#define INF 0x3f3f3f3f
#define st first
#define nd second
#define DEBUG(x) printf("%d\n", x), exit(0);
// LLONG_MIN LLONG_MAX INT_MIN INT_MAX

int inp[202020];
int odd, even;
set <int> S;
vector <int> idx;
int n, m;

int main(){
    scanf("%d %d", &n, &m);
    for(int i = 0; i < n; i++){
        scanf("%d", &inp[i]);
        if(S.count(inp[i]) || (inp[i] % 2 == 1 && odd >= n / 2) || (inp[i] % 2 == 0 && even >= n / 2)){
            idx.pb(i);
        }
        else{
            (inp[i] % 2) ? odd++ : even++;
            S.insert(inp[i]);
        }
    }
    int ans = idx.size();
    for(int i = 1; i <= m; i++){
        if(idx.empty())
            break;
        if(S.count(i) == 0){
            if(i % 2 == 1 && odd < n / 2){
                inp[idx.back()] = i;
                odd++;
                idx.pop_back();
            }
            else if(i % 2 == 0 && even < n / 2){
                inp[idx.back()] = i;
                even++;
                idx.pop_back();
            }
        }
    }
    if(idx.size())
        printf("-1\n");
    else{
        printf("%d\n", ans);
        for(int i = 0; i < n; i++){
            printf("%d%c", inp[i], (i == n - 1) ? '\n' : ' ');
        }
    }
    return 0;
}
