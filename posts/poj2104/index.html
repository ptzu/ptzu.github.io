POJ 2104 K-th Number

Jan 28, 2018
4 min read
POJ

題目

input

給兩正整數 n, m, 1 <= n <= 100000, 1 <= m <= 5000 代表 n 個數字, m 個查詢 第二列輸入 n 個不超過 10^9 的數字 接著每個查詢輸入 l, r, k 代表問區間 [l, r] 的第 k 大數字

題解

給一序列, 然後不斷問不同區間的第 k 大數字 目前看到的作法有 歸併樹、劃分樹、可持久化線段樹(主席樹)、平方分解

作法

離散化 + 主席樹

AC Code

#include <cstdio>
#include <algorithm>
#include <vector>

using namespace std;

typedef pair <int,int> pii;
typedef long long ll;

#define pb push_back
#define sz(a) (int)a.size()
#define INF 0x3f3f3f3f
#define st first
#define nd second
#define fio ios::sync_with_stdio(false), cin.tie(0)
#define DEBUG(x) printf("Here: %d\n", x), exit(0);
// LLONG_MIN LLONG_MAX INT_MIN INT_MAX
const int MAX_N = 100000;

struct node{
    int l, r, num;
}seg[MAX_N * 40];
int root[MAX_N + 10], cnt;
int a[MAX_N + 10];
vector <int> pool;

void update(int l, int r, int &now, int k){
    cnt++;
    seg[cnt] = seg[now]; // 複製節點
    seg[cnt].num++; // 更新節點
    now = cnt;
    if(l + 1 == r) return ;
    int m = (l + r) / 2;
    if(m > k) update(l, m, seg[now].l, k);
    else update(m, r, seg[now].r, k);
}

int query(int l, int r, int a, int b, int k){
    if(l + 1 == r) return l;
    int m = (l + r) / 2;
    int num = seg[seg[b].l].num - seg[seg[a].l].num;
    if(num > k) query(l, m, seg[a].l, seg[b].l, k);
    else query(m, r, seg[a].r, seg[b].r, k - num);
}

int main(){
    int n, m;
    scanf("%d%d", &n, &m);
    for(int i = 0; i < n; i++) scanf("%d", &a[i]), pool.pb(a[i]);
    sort(pool.begin(), pool.end());
    pool.resize(unique(pool.begin(), pool.end()) - pool.begin()); // 離散化

    for(int i = 1; i <= n; i++){
        int pos = lower_bound(pool.begin(), pool.end(), a[i - 1]) - pool.begin();
        root[i] = root[i - 1]; // 一開始先從上個版本的 root 走
        update(0, n, root[i], pos);
    }
    for(int i = 0; i < m; i++){
        int l, r, k;
        scanf("%d%d%d", &l, &r, &k);
        k--;
        int idx = query(0, n, root[l - 1], root[r], k);
        printf("%d\n", pool[idx]);
    }
    return 0;
}

作法

歸併樹 + 二分搜 歸併樹即為在建樹過程中, 使用 merge sort 的概念 由於每個節點都是排序好的序列 我們就可以在某個區間去查找不大於 x 的數量 最後在用 01 型二分搜, 判斷這個數的數量 在區間是否 >= k 找到的 upper_bound 即為第 k 大

AC code

#include <cstdio>
#include <algorithm>
#include <vector>

using namespace std;

typedef pair <int,int> pii;
typedef long long ll;

#define pb push_back
#define sz(a) (int)a.size()
#define INF 0x3f3f3f3f
#define st first
#define nd second
#define fio ios::sync_with_stdio(false), cin.tie(0)
#define DEBUG(x) printf("Here: %d\n", x), exit(0);
// LLONG_MIN LLONG_MAX INT_MIN INT_MAX
const int MAX_N = 100000;

int N, M, SN;
vector <int> seg[MAX_N * 4 + 10];
int a[MAX_N + 10];
int sorted[MAX_N + 10];

void seg_init(){
    SN = 1;
    while(SN < N) SN <<= 1;
}

void seg_build(int idx, int l, int r){
    if(l + 1 == r){
        if(l < N) seg[idx].pb(a[l]);
        return ; // 葉節點一定要 return
    }
    int m = (l + r) / 2;
    seg_build(idx * 2 + 1, l, m);
    seg_build(idx * 2 + 2, m, r);
    seg[idx].resize(r - l);
    merge(seg[idx * 2 + 1].begin(), seg[idx * 2 + 1].end(),
          seg[idx * 2 + 2].begin(), seg[idx * 2 + 2].end(), seg[idx].begin());
}

int seg_query(int a, int b, int x, int idx, int l, int r){
    if(r <= a || l >= b) return 0;
    if(a <= l && r <= b){
        return upper_bound(seg[idx].begin(), seg[idx].end(), x) - seg[idx].begin();
    }
    int m = (l + r) / 2;
    int res = 0;
    res += seg_query(a, b, x, idx * 2 + 1, l, m);
    res += seg_query(a, b, x, idx * 2 + 2, m, r);
    return res;
}

int kth(int a, int b, int k){
    int lb = -1, ub = N - 1;
    while(ub - lb > 1){
        int mid = (ub + lb) / 2;
        if(seg_query(a, b, sorted[mid], 0, 0, SN) >= k) ub = mid;
        else lb = mid;
    }
    return sorted[ub];
}

int main(){
    scanf("%d%d", &N, &M);
    for(int i = 0; i < N; i++) scanf("%d", &a[i]);
    copy(a, a + N, sorted);
    sort(sorted, sorted + N);
    seg_init();
    seg_build(0, 0, SN);
    for(int i = 0; i < M; i++){
        int l, r, k;
        scanf("%d%d%d", &l, &r, &k);
        l--;r--;
        printf("%d\n", kth(l, r + 1, k));
    }
    return 0;
}
