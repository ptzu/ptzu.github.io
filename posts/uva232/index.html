Uva 232 - Crossword Answers

Oct 22, 2017
2 min read
Uva Implement

題目

input

1 <= r,c <= 10

題解

給一個 r * c 的格子, 有黑白格, 黑格用 * 表示 一個白格的上方 or 右方若 ‘沒有格子’ 或 ‘有黑色格子’ 則稱該白格為 eligible 將所有 eligible 格子從左到右, 上到下依序編號 每個單字的開頭一定要放在 eligible 求 Across(橫向單字), Down(縱向單字) 有哪些

作法

題目非常長, 不過作法卻很簡單, 實作就好了

AC code

#include <bits/stdc++.h>

using namespace std;

typedef pair <int,int> pii;
typedef long long ll;

#define pb push_back
#define sz(a) (int)a.size()
#define INF 0x3f3f3f3f
#define st first
#define nd second
#define fio ios::sync_with_stdio(false), cin.tie(0)
#define DEBUG(x) printf("Here: %d\n", x), exit(0);
// LLONG_MIN LLONG_MAX INT_MIN INT_MAX
//const int MAX_N = 101010;

string a[15];
bool vis[15][15];
int R, C;

struct point{
    int idx, r, c;
};

vector <point> v;

void init(){

}

bool isEligible(int r, int c){
    pii dir[] = {{0, -1}, {-1, 0}};
    for(int i = 0; i < 2; i++){
        int rr = r + dir[i].st;
        int cc = c + dir[i].nd;
        if(rr < 0 || cc < 0 || a[rr][cc] == '*')
            return true;
    }
    return false;
}

int main(){
    int TC = 0;
    while(scanf("%d%d\n", &R, &C) != EOF){
        v.clear();
        TC++;
        if(R == 0) break;
        if(TC > 1) puts("");
        printf("puzzle #%d:\n", TC);

        for(int i = 0; i < R; i++)
            getline(cin, a[i]);
        int cnt = 1;
        for(int i = 0; i < R; i++){
            for(int j = 0; j < C; j++){
                if(a[i][j] == '*') continue;
                if(isEligible(i, j)) v.pb((point){cnt++, i, j});
            }
        }
        memset(vis, 0, sizeof(vis));
        printf("Across\n");
        for(point p : v){
            if(vis[p.r][p.c]) continue;
            if(p.idx < 10) printf("  %d.", p.idx);
            else printf(" %d.", p.idx);

            while(p.c < C){
                if(a[p.r][p.c] == '*') break;
                vis[p.r][p.c] = true;
                printf("%c", a[p.r][p.c]);
                p.c++;
            }
            puts("");
        }
        memset(vis, 0, sizeof(vis));
        printf("Down\n");
        for(point p : v){
            if(vis[p.r][p.c]) continue;
            if(p.idx < 10) printf("  %d.", p.idx);
            else printf(" %d.", p.idx);

            while(p.r < R){
                if(a[p.r][p.c] == '*') break;
                vis[p.r][p.c] = true;
                printf("%c", a[p.r][p.c]);
                p.r++;
            }
            puts("");
        }
    }
    return 0;
}
