bool Suf(string a, string b){
    if(a.length() > b.length()) return false;
    return a == b.substr(b.length() - a.length(), a.length());
}

Graph linked list

const int MAX_N = 50010;
struct node{
    int to, w;
    node* next;
}*g[MAX_N], edges[MAX_N * 3];
int tot;

inline void addEdge(int a, int b, int c){
    node *e = &edges[tot++];
    e->to = b;
    e->w = c;
    e->next = g[a];
    g[a] = e;
}
