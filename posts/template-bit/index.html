注意是 1-based index

Code

const int MAX_N = 20000;
int bit[MAX_N + 10];

int add(int i, int x){
    while(i <= n){
        bit[i] += x;
        i += i & -i;
    }
}

int sum(int i){
    int ans = 0;
    while(i){
        ans += bit[i];
        i -= i & -i;
    }
    return ans;
}
