String 切 Token

Jan 12, 2018
1 min read
String

有兩種方法

  1. 將 string 轉為 c 字串, 然後用 strtok
  2. 利用 stringstream

第一種：

#include <bits/stdc++.h>

using namespace std;

string s;

int main(){
    while(getline(cin, s)){
        char cstr[10000];
        strcpy(cstr, s.c_str());
        char *token = strtok(cstr, " ");
        while(token != NULL){
            printf("%s ", token);
            token = strtok(NULL, " ");
        }
        puts("");
    }
    return 0;
}

第二種:

#include <bits/stdc++.h>

using namespace std;

string s;
stringstream ss;

int main(){
    while(getline(cin, s)){
        ss.str("");
        ss.clear();
        // 上面兩行是清空 stringstream, 兩行都必加
        ss << s;
        int n;
        while(ss >> n){
            cout << n << " ";
        }
        cout << endl;
    }
    return 0;
}
