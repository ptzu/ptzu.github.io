POJ 2236 Wireless Network

Aug 9, 2016
2 min read
POJ dsu

題解

有 N 台壞掉的電腦, 要使他們能重新連通 A 和 B 能連通必須在距離 d 之內, 不過可以透過第三者連接 input 會給每台電腦座標 接著有兩種指令：

  1. O a: 代表修復 a 電腦
  2. S a b: 代表測試 a 和 b 是否連通

作法

用 bool 陣列紀錄哪台電腦已經修復 接著迴圈掃其他電腦, 假如距離 < d 且該電腦已修復 就將他們加到同個集合(併查集, Disjoint Set Union) 當要查詢時就能直接看他們是否在同一個集合啦

#include <cstdio>
#include <cstdlib>
#include <cmath>

using namespace std;

typedef pair<int,int> pii;
typedef long long ll;

#define pb push_back
#define pf push_front
#define mp make_pair
#define sz(a) (int)a.size()
#define i128 __int128
#define INF 0x3f3f3f3f
#define st first
#define nd second
// LLONG_MIN LLONG_MAX INT_MIN INT_MAX

int n, d;
int pa[1001 + 10];
bool work[1001 + 10];
pii p[1001 + 10];

void init(){
    for(int i = 1; i <= n; i++)
        pa[i] = i;
}

int find(int x){
    if(pa[x] == x)
        return x;
    return pa[x] = find(pa[x]);
}

void unite(int a, int b){
    a = find(a);
    b = find(b);

    if(a == b)
        return ;
    pa[a] = pa[b];
}

int square(int x){
    return x * x;
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);

    cin >> n >> d;
    init();
    for(int i = 1; i <= n; i++){
        cin >> p[i].st >> p[i].nd;
    }
    string s;
    while(cin >> s){
        if(s[0] == 'O'){
            int a;
            cin >> a;
            work[a] = true;
            for(int i = 1; i <= n; i++){
                if(i == a)
                    continue;
                if(work[i] && square(p[a].st - p[i].st) + square(p[a].nd - p[i].nd) <= square(d))
                    unite(a, i);
            }
        }
        else{
            int a, b;
            cin >> a >> b;
            a = find(a);
            b = find(b);

            if(a == b)
                cout << "SUCCESS\n";
            else
                cout << "FAIL\n";
        }
    }
    return 0;
}
