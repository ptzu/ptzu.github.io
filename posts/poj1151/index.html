POJ 離散化 掃描線

題目

input

1 <= n <= 100 0 <= x1 < x2 <= 100000, 0 <= y1 < y2 <= 100000

題解

給 n 個矩形, 求所有矩形覆蓋面積

作法

座標範圍有點大, 先離散化 y 軸 然後掃描線掃 x 軸 若掃到某個矩形的 x1, 就將 y1 ~ y2 區間 +1, 反之掃到 x2 就 -1 以每一段的 x 座標為一個區段, 則這一個區段的面積就是 (x軸變化 * y軸被覆蓋的長度) 將所有區段加起來就是答案

AC code

#include <cstdio>
#include <algorithm>
#include <cstring>
#include <vector>

using namespace std;

typedef pair <int,int> pii;
typedef long long ll;

#define pb push_back
#define sz(a) (int)a.size()
#define INF 0x3f3f3f3f
#define st first
#define nd second
#define fio ios::sync_with_stdio(false), cin.tie(0)
#define DEBUG(x) printf("Here: %d\n", x), exit(0);
// LLONG_MIN LLONG_MAX INT_MIN INT_MAX
const int MAX_N = 10000;

struct Node{
    double pos;
    int d, u, type;
    bool operator <(const Node& a)const{
        if(pos == a.pos) return type > a.type;
        return pos < a.pos;
    }
};

int n, SN;
double x[2 * MAX_N + 10], y[2 * MAX_N + 10];
vector <double> yPool;
int seg[4 * MAX_N + 10];
double cnt[4 * MAX_N + 10];
vector <Node> sweep;

void discrete(){
    sort(yPool.begin(), yPool.end());
    yPool.resize(unique(yPool.begin(), yPool.end()) - yPool.begin());
}

void init(){
    yPool.clear();
    memset(seg, 0, sizeof(seg));
    memset(cnt, 0, sizeof(cnt));
    sweep.clear();
}

void seg_init(int n){
    SN = 1;
    while(SN < n)
        SN <<= 1;
}

void seg_maintain(int k, int l, int r){
    if(seg[k]){
        cnt[k] = yPool[r] - yPool[l];
    }
    else{
        cnt[k] = cnt[k * 2 + 1] + cnt[k * 2 + 2];
    }
}

void seg_update(int a, int b, int x, int k, int l, int r){
    if(l >= b || r <= a) return ;
    if(a <= l && r <= b){
        seg[k] += x;
        // cout << k << endl;
        seg_maintain(k, l, r);
        return ;
    }
    int m = (l + r) / 2;
    seg_update(a, b, x, k * 2 + 1, l, m);
    seg_update(a, b, x, k * 2 + 2, m, r);
    seg_maintain(k, l, r);
}

int main(){
    int TC = 1;
    while(scanf("%d", &n) && n){
        init();
        for(int i = 0; i < n; i++){
            scanf("%lf%lf%lf%lf", &x[i], &y[i], &x[i + n], &y[i + n]);
            yPool.pb(y[i]);
            yPool.pb(y[i + n]);
        }
        discrete();
        for(int i = 0; i < n; i++){
            int y1 = lower_bound(yPool.begin(), yPool.end(), y[i]) - yPool.begin();
            int y2 = lower_bound(yPool.begin(), yPool.end(), y[i + n]) - yPool.begin();
            sweep.pb((Node){x[i], y1, y2, 1});
            sweep.pb((Node){x[i + n], y1, y2, -1});
        }
        sort(sweep.begin(), sweep.end());
        double ans = 0;
        // y 離散化後的數量, 可能為 2 * n
        seg_init(sz(yPool));
        seg_update(sweep[0].d, sweep[0].u, sweep[0].type, 0, 0, SN + 1);
        for(int i = 1; i < sz(sweep); i++){
            ans += (sweep[i].pos - sweep[i - 1].pos) * cnt[0];
            seg_update(sweep[i].d, sweep[i].u, sweep[i].type, 0, 0, SN + 1);
        }
        printf("Test case #%d\n", TC++);
        printf("Total explored area: %.2f\n", ans);
        puts("");
    }
    return 0;
}
