#include <bits/stdc++.h>

using namespace std;
typedef pair <int,int> pii;
typedef long long ll;

#define pb push_back
#define sz(a) (int)a.size()
#define INF 0x3f3f3f3f
#define st first
#define nd second
#define fio ios::sync_with_stdio(false), cin.tie(0)
#define DEBUG(x) printf("Here: %d\n", x), exit(0);
// LLONG_MIN LLONG_MAX INT_MIN INT_MAX
const int mod = 1e9 + 7;

string s;
int k;
int dp[1010];
int C[1010][1010];


int main(){
    fio;
    cin >> s >> k;
    if(k == 0){
        puts("1");
        return 0;
    }
    dp[1] = 0;
    for(int i = 2; i <= 1000; i++)
        dp[i] = dp[__builtin_popcount(i)] + 1;
    C[0][0] = 1;
    for(int i = 1; i <= 1000; i++){
        C[i][0] = 1;
        for(int j = 1; j <= 1000; j++){
            C[i][j] = (C[i - 1][j] + C[i - 1][j - 1]) % mod;
        }
    }
    int one_num = 0, len = s.length();
    int ans = 0;
    for(int i = 0; i < len; i++){
        if(s[i] == '1'){
            for(int j = max(one_num, 1); j <= 1000; j++){
                if(dp[j] == k - 1)
                    ans = (ans + C[len - i - 1][j - one_num]) % mod;
            }
            one_num++;
        }
    }
    if(dp[one_num] == k - 1) ans = (ans + 1) % mod; // 這個不會在迴圈計算
    // 當只有一個 bit 1 的組合, 必須扣掉單純為 1, 因為他已經是結束狀態
    if(k == 1) ans = (ans - 1) % mod;
    printf("%d\n", ans);
    return 0;
}
