POJ 2421 Constructing Roads

Jan 12, 2018
2 min read
POJ Graph MST

題目

input

第一行輸入正整數 N, 3 <= N <= 100 代表有 N 個村莊 接著輸入一個 N * N 的相鄰矩陣, 代表邊 然後輸入一個正整數 q 接著 q 行輸入 a, b 代表 a, b 之間已經有建好路

題解

建路需要成本, 已經有些路建好 求要連接所有村莊, 最小成本是多少？

作法

求 MST, 不過這題範圍比較小 所以我拿來測試 Sollin 演算法

AC code

#include <cstdio>
#include <algorithm>
#include <climits>
#include <cstring>

using namespace std;
typedef pair <int,int> pii;
typedef long long ll;

#define pb push_back
#define sz(a) (int)a.size()
#define INF 0x3f3f3f3f
#define st first
#define nd second
#define fio ios::sync_with_stdio(false), cin.tie(0)
#define DEBUG(x) printf("Here: %d\n", x), exit(0);
// LLONG_MIN LLONG_MAX INT_MIN INT_MAX
// const int MAX_V = 100100;

struct edge{
    int u, v, w;
};

struct Union_Find {
    int par[1010], num[1010];
    void init(int n){
        for(int i = 0; i < n; i++){
            par[i] = i;
            num[i] = 1;
        }
    }
    int find(int x){
        if(x == par[x])
            return x;
        else
            return par[x] = find(par[x]);
    }
    void unite(int a, int b){
        a = find(a);
        b = find(b);
        if(a == b)
            return ;
        if(num[a] > num[b])
            swap(a, b);
        par[a] = b;
        num[b] += num[a];
    }
}U;

edge edges[100000];
int V, E;
int a[1000][1000];
int mn[1000];
int ComponentSize;

int sollin(){
    int ans = 0;
    while(ComponentSize > 1){
        for(int i = 0; i < V; i++) mn[i] = INT_MAX;
        for(int i = 0; i < E; i++){
            edge e = edges[i];
            int a = U.find(e.u);
            int b = U.find(e.v);
            if(a == b) continue;
            if(e.w < mn[a]) mn[a] = e.w;
            if(e.w < mn[b]) mn[b] = e.w;
        }
        for(int i = 0; i < E; i++){
            edge e = edges[i];
            int a = U.find(e.u);
            int b = U.find(e.v);
            if(a == b) continue;
            if(e.w == mn[a] && e.w == mn[b]){
                U.unite(a, b);
                ans += e.w;
                ComponentSize--;
            }
        }
    }
    return ans;
}

int main(){
    scanf("%d", &V);
    memset(edges, 0, sizeof(edges));
    int idx = 0;
    for(int i = 0; i < V; i++){
        for(int j = 0; j < V; j++){
            scanf("%d", &a[i][j]);
            if(i > j){
                edges[idx++] = {i, j, a[i][j]};
            }
        }
    }
    E = idx;
    ComponentSize = V;
    U.init(V);
    int q;
    scanf("%d", &q);
    for(int i = 0; i < q; i++){
        int u, v;
        scanf("%d%d", &u, &v);
        u--;v--;
        if(U.find(u) != U.find(v)){
            U.unite(u, v);
            ComponentSize--;
        }
    }
    printf("%d\n", sollin());
    return 0;
}
