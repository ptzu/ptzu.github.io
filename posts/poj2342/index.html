題解

派對中有 n 個員工, 老闆不希望對於某個員工, 他的上司和下屬出現在派對, 每個人都有一個歡樂值, 請問派對中最大的歡樂值總和為多少？

作法

樹型DP, bottom-up dp[i][0] 表第 i 個員工不出席的歡樂值 dp[i][1] 表第 i 個員工出席的歡樂值

AC code

#include <cstdio>
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

typedef pair <int,int> pii;
typedef long long ll;

#define pb push_back
#define sz(a) (int)a.size()
#define INF 0x3f3f3f3f
#define st first
#define nd second
#define fio ios::sync_with_stdio(false), cin.tie(0)
// #define DEBUG(x) printf("Here: %d\n", x), exit(0);
// LLONG_MIN LLONG_MAX INT_MIN INT_MAX

vector <int> g[6060];
int conviviality[6060];
bool point[6060] ;// 判斷是否有被指向
int n;
int dp[6060][2];

void DFS(int x){
    for(int i = 0; i < sz(g[x]); i++){
        int son = g[x][i];
        DFS(son);
        dp[x][0] += max(dp[son][0], dp[son][1]);
        dp[x][1] += dp[son][0];
    }
}

int main(){
    // fio;
    // #ifdef DEBUG
    // printf ("TEST\n");
    // #else
    // printf("FAIL\n");
    // #endif

    cin >> n;
    for(int i = 0; i < n; i++){
        scanf("%d", &conviviality[i]);
        dp[i][1] = conviviality[i];
    }
    for(int i = 0; i < n; i++){
        int u, v;
        scanf("%d %d", &u, &v);
        u--;v--;
        g[v].pb(u);
        point[u] = true;
    }
    //　找root
    int root;
    for(int i = 0; i < n; i++){
        if(!point[i])
            root = i;
    }
    DFS(root);
    printf("%d\n", max(dp[root][0], dp[root][1]));
    return 0;
}
